vector<int> bfs(int start, int n) {

            int path2 = min(distFromT[b2U], distFromT[b2V]) + 1;  // ‡∏Å‡∏•‡∏±‡∏ö‡∏ö‡πâ‡∏≤‡∏ô‡∏ú‡πà‡∏≤‡∏ô‡∏™‡∏∞‡∏û‡∏≤‡∏ô‡∏ó‡∏µ‡πà‡∏™‡∏≠‡∏á
